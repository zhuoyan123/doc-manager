## 动态规划问题解题套路框架
首先动态规划问题的一般形式就是求最值，比如求最长递增子序列，最小编辑距离等等。

动态规划三要素：重叠子问题、最优子结构、状态转移方程。

明确base case -> 明确状态 -> 明确选择 -> 定义dp数组/函数的定义。可以套这个框架：
```JavaScript
// 初始化base case
dp[0][0][...] = base;
// 进行状态转移
for (状态1 in 状态1的所有取值) {
    for (状态2 in 状态2的所有取值) {
        for ... {
            dp[状态1][状态2][...] = 求最值(选择1，选择2，...);
        }
    }
}
```

### 一、斐波那契数列
```typescript
function fib(n: number): number {
    if (n === 0 || n === 1) return n;
    let dp1 = 0;
    let dp2 = 1;
    for (let i = 2; i <= n; i++) {
        const dp = dp1 + dp2;
        dp1 = dp2;
        dp2 = dp;
    }
    return dp2;
}
```

### 二、零钱兑换
```typescript
function coinChange(coins: number[], amount: number): number {
    if (amount === 0) return 0;
    const dp = (new Array(amount + 1)).fill(amount + 1);
    dp[0] = 0;
    for (let i = 1; i <= amount; i++) {
        for (let coin of coins) {
            if (i - coin < 0) continue;
            const d = 1 + dp[i - coin];
            dp[i] = Math.min(dp[i], d);
        }
    }
    return (dp[amount] === amount + 1 ? -1 : dp[amount]);
}
```

## 回朔算法解题套路框架
解决一个回溯问题，实际上就是一个决策树的遍历过程，你只需要考虑3个问题：

1. 路径：也就是已经做出的选择。
2. 选择列表：也就是你当前可以做的选择。
3. 结束条件：也就是达到决策树底层，无法再做选择的条件。

回溯算法框架：
```JavaScript
result = [];
function backtrack(路径，选择列表) {
    if (满足结束条件) {
        result.push(路径);
        return;
    }
    for (let 选择 of 选择列表) {
        做选择
        backtrack(路径，选择列表);
        撤销选择
    }
}
```
其核心就是for循环里面的递归在递归调用之前做选择，在递归调用之后撤销选择

```typescript
function permute(nums: number[]): number[][] {
    const res = [];
    function backtrack(nums: number[], track: number[], used: boolean[]) {
        if (track.length === nums.length) {
            res.push([...track]);
            return;
        }
        for (let i = 0; i < nums.length; i++) {
            if (used[i]) {
                continue;
            }
            track.push(nums[i]);
            used[i] = true;
            backtrack(nums, track, used);
            track.pop();
            used[i] = false;
        }
    }
    backtrack(nums, [], []);
    return res;
};
```

### 二、N皇后问题
```typescript
function solveNQueens(n: number) {
    const res = [];
    const board = new Array(n).fill(new Array(n).fill('.').join(''));
    backtrack(board, 0);

    function backtrack(board: string[], row: number) {
        if (row === board.length) {
            res.push([...board]);
            return;
        }
        let n = board[row].length;
        for (let col = 0; col < n; col++) {
            if (!isValid(board, row, col)) {
                continue;
            }
            const rowArray = board[row].split('');
            rowArray[col] = 'Q';
            board[row] = rowArray.join('');
            backtrack(board, row + 1);
            rowArray[col] = '.';
            board[row] = rowArray.join('');
        }
    }

    function isValid(board: string[], row: number, col: number): boolean {
        const m = board.length;
        for (let i = 0; i < n; i++) {
            if (board[i][col] === 'Q') {
                return false;
            }
        }
        for (let i = row - 1, j = col + 1; i >= 0 && j < n; i--, j++) {
            if (board[i][j] === 'Q') {
                return false;
            }
        }
        for (let i = row - 1, j = col - 1; i >= 0 && j >= 0; i--, j--) {
            if (board[i][j] === 'Q') {
                return false;
            }
        }
        return true;
    }
    return res;
};
```

## BFS算法（广度优先遍历）
问题的本质就是让你在一副「图」中找到从起点start到终点target的最短距离，这个例子听起来很枯燥，但是BFS算法问题其实都是在干这个事儿。
```typescript
interface Node {
    value: unkonwn;
    next: Node;
}
function bfs(start: Node, target: Node) {
    const q = [];      // 核心数据结构
    const visited = [];  //避免走回头路
    visited.push(start);
    let step = 0;
    while (q.length !== 0) {
        let sz = q.length;
        // 将当前队列中的所有节点向四周扩散
        for (let i = 0; i < sz; i++) {
            const cur = q.shift();
            // 划重点，这里判断是否达到终点
            if (Object.is(cur, target)) {
                return step;
            }
            // 将cur的相邻节点加入队列
            for (let x of cur的相邻节点) {
                if (!visited.includes(x)) {
                    q.push(x)
                    visited.push(x);
                }
            }
        }
        step++;
    }
}
```

### 一、二叉树的最小深度
```typescript
class TreeNode {
    val: number
    left: TreeNode | null
    right: TreeNode | null
    constructor(val?: number, left?: TreeNode | null, right?: TreeNode | null) {
        this.val = (val===undefined ? 0 : val)
        this.left = (left===undefined ? null : left)
        this.right = (right===undefined ? null : right)
    }
}

function minDepth(root: TreeNode | null): number {
    if (root == null) return 0;
    let queue = [];
    let depth = 0;
    queue.push(root);
    while (queue.length !== 0) {
        const sz = queue.length;
        for (let i = 0; i < sz; i++) {
            const cur = queue.shift();
            if (cur.left === null && cur.right === null) {
                return depth;
            }
            if (cur.left) {
                queue.push(cur.left);
            }
            if (cur.right) {
                queue.push(cur.right);
            }
        }
        depth++;
    }
};
```

### 二、解开密码锁的最少次数（双向bfs）
```typescript
function plusOne(s: string, j: number) {
    const ch = s.split('');
    if (ch[j] === '9') {
        ch[j] = '0';
    } else {
        ch[j] = String(parseInt(ch[j]) + 1);
    }
    return ch.join('');
}

function minusOne(s: string, j: number) {
    const ch = s.split('');
    if (ch[j] === '0') {
        ch[j] = '9';
    } else {
        ch[j] = String(parseInt(ch[j]) - 1);
    }
    return ch.join('');
}

function openLock(deadends: string[], target: string): number {
    // 记录需要跳过的死亡密码
    const deads = new Set();
    for (let s of deadends) {
        deads.add(s);
    }
    // 记录已经穷举过的密码，防止走回头路
    const visited = new Set();
    let q1: Set<string> = new Set();
    let q2: Set<string> = new Set();
    q1.add('0000');
    q2.add(target)
    let step = 0;
    while (q1.size && q2.size) {
        // 哈希集合在遍历的过程中不能修改，用temp存储扩散结果。
        const temp: Set<string> = new Set();
        // 将q1中的所有节点向四周扩散
        for (let cur of q1) {
            // 判断是否到达终点
            if (deads.has(cur)) {
                continue;
            }
            if (q2.has(cur)) {
                return step;
            }
            visited.add(cur);
            // 将一个节点未遍历的相邻节点加入队列
            for (let j = 0; j < 4; j++) {
                const up = plusOne(cur, j);
                const down = minusOne(cur, j);
                if (!visited.has(up)) {
                    temp.add(up);
                }
                if (!visited.has(down)) {
                    temp.add(down);
                }
            }
        }
        step++;
        q1 = q2;
        q2 = temp;
    }
    return -1;
}
```
